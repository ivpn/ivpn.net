BreadcrumbList – hierarchical pages only, no language or catch-all sections */ -}} {{- $base := .Site.BaseURL -}} {{- $orgID := printf "%s#organization" $base -}} {{- $siteID := printf "%s#website" $base -}} {{- /* The English/Spanish landing pages (Kind=page, URL=/en/ or /es/) act as home */ -}} {{- $isLanding := or .IsHome (eq .Permalink .Site.Home.Permalink) -}} {{- /* ── 1. Organization (same on every page) ── */ -}} {{- $org := dict "@type" "Organization" "@id" $orgID "name" "IVPN" "legalName" "IVPN Limited" "url" $base "foundingDate" "2009" "logo" (dict "@type" "ImageObject" "@id" (printf "%s#logo" $base) "url" (printf "%simages-static/uploads/og_1200-627.png" $base) "contentUrl" (printf "%simages-static/uploads/og_1200-627.png" $base) "width" 1200 "height" 627 ) "description" "Audited, open-source VPN service with WireGuard, killswitch and tracker blocker. "areaServed" "Worldwide" "contactPoint" (dict "@type" "ContactPoint" "email" "support@ivpn.net" "contactType" "customer support" "availableLanguage" (slice "English" "Spanish") ) "sameAs" (slice "https://twitter.com/ivpnnet" "https://github.com/ivpn" "https://www.reddit.com/r/IVPN/" "https://mastodon.social/@ivpn" ) -}} {{- /* ── 2. WebSite (same on every page) ── */ -}} {{- $website := dict "@type" "WebSite" "@id" $siteID "url" $base "name" "IVPN" "publisher" (dict "@id" $orgID) -}} {{- /* ── 3. Page-specific node ── */ -}} {{- $pageNode := dict -}} {{- if $isLanding -}} {{- /* Landing page: Organization + WebSite are the content; no extra page node */ -}} {{- else if and (eq .Type "blog") (eq .Kind "page") -}} {{- /* Individual blog post → BlogPosting */ -}} {{- $authorList := slice -}} {{- range .Params.authors -}} {{- $authorList = $authorList | append (dict "@type" "Person" "name" .) -}} {{- end -}} {{- if not $authorList -}} {{- $authorList = slice (dict "@type" "Organization" "name" "IVPN") -}} {{- end -}} {{- $pageNode = dict "@type" "BlogPosting" "headline" .Title "url" .Permalink "datePublished" (.Date.Format "2006-01-02T15:04:05Z07:00") "dateModified" (.Lastmod.Format "2006-01-02T15:04:05Z07:00") "author" $authorList "publisher" (dict "@type" "Organization" "@id" $orgID "name" "IVPN") "mainEntityOfPage" (dict "@type" "WebPage" "@id" .Permalink) -}} {{- with .Description -}} {{- $pageNode = merge $pageNode (dict "description" .) -}} {{- end -}} {{- with .Params.images -}} {{- $pageNode = merge $pageNode (dict "image" (index . 0)) -}} {{- end -}} {{- else -}} {{- /* Everything else → WebPage (section list pages, knowledgebase, guides…) */ -}} {{- $pageNode = dict "@type" "WebPage" "@id" (printf "%s#webpage" .Permalink) "name" .Title "url" .Permalink "inLanguage" .Site.Language.Lang "isPartOf" (dict "@id" $siteID) "publisher" (dict "@id" $orgID) -}} {{- with .Description -}} {{- $pageNode = merge $pageNode (dict "description" .) -}} {{- end -}} {{- end -}} {{- /* ── 4. BreadcrumbList ── */ -}} {{- /* Skip: landing pages, and pages with no title (e.g. headless app pages) */ -}} {{- $crumbNode := dict -}} {{- if and (not $isLanding) .Title -}} {{- /* The catch-all content-organisation folder (/en/pages/, /es/pages/) must not appear as a trail item. Sub-sections with URL overrides (knowledgebase/billing/, privacy-guides/, …) have different Permalinks so they are kept. */ -}} {{- $langPagesPath := printf "/%s/pages/" $.Site.Language.Lang -}} {{- $crumbs := slice (dict "@type" "ListItem" "position" 1 "name" "Home" "item" .Site.Home.Permalink) -}} {{- $pos := 2 -}} {{- range collections.Reverse .Ancestors -}} {{- $skipAncestor := or .IsHome (eq .RelPermalink $langPagesPath) -}} {{- if and (not $skipAncestor) .Title -}} {{- $crumbs = $crumbs | append (dict "@type" "ListItem" "position" $pos "name" .Title "item" .Permalink) -}} {{- $pos = add $pos 1 -}} {{- end -}} {{- end -}} {{- $crumbs = $crumbs | append (dict "@type" "ListItem" "position" $pos "name" .Title "item" .Permalink) -}} {{- $crumbNode = dict "@type" "BreadcrumbList" "@id" (printf "%s#breadcrumb" .Permalink) "itemListElement" $crumbs -}} {{- end -}} {{- /* ── Assemble single @graph and emit ── */ -}} {{- $graph := slice $org $website -}} {{- if $pageNode -}}{{- $graph = $graph | append $pageNode -}}{{- end -}} {{- if $crumbNode -}}{{- $graph = $graph | append $crumbNode -}}{{- end -}} {{- dict "@context" "https://schema.org" "@graph" $graph | jsonify (dict "indent" " " "escapeHTML" false) | printf "" | safeHTML -}}
